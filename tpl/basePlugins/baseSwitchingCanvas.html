tab切换标签--内容区域框(Canvas .tabbable-custom ) 使用注意事项
  1. var echarts = require('echarts');//引入图形
  2. 例： 
         afterRender:function(){
        //在页面初始化的时候第一个tab的可见区域中的图像是调用这个showSchoolSafe0()方法生成的图像所以要在这里调用
         this.showSchoolSafe0();
     }
    //tab切换事件点击后应有对应的方法
    events: {
       //tab切换事件
       "click #canvas0": "canvas0",
       "click #canvas1": "canvas1",
        },
     //对应点击时的方法调用下面的数据生成图像
        //因为第一个图是在初始化的时候就已经是可见区域了所以不用延迟就可以进行绘制canvas
        canvas0: function () {
            this.showSchoolSafe0();
        },
        //初始化的时候第二个区域是不可见的，canvas不能绘制图形，所以要先让其可见后再进行canvas绘制，顾给此方法增加了定时器延迟执行200毫秒
        canvas1: function () {
            setTimeout(function () {
                _this.showSchoolSafe1();
            }, 200)
        },
    //以下是对应的canvas数据结构
        showSchoolSafe0: function (){
        //第一张canvas的数据及结构
            }
        showSchoolSafe1: function (){
        //第二张canvas的数据及结构
           }
